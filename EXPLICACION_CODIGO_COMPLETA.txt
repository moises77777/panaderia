<?php

namespace App\Http\Controllers;
// Espacio de trabajo para evitar conflictos de nombres
// Organiza las clases por carpetas

use App\Models\Empleado;
use App\Models\InfoEmpleado;
use Illuminate\Http\Request;
// Importamos las clases que vamos a usar

class EmpleadoController extends Controller
{
    // Hereda de Controller que tiene métodos básicos de Laravel

    public function index()
    {
        // Método para LISTAR todos los empleados
        // Se accede con GET /api/empleados
        
        try {
            // Intenta ejecutar este código
            $empleados = Empleado::with(['compras', 'ventas', 'infoEmpleado'])->get();
            // Empleado::with() - Carga relaciones para evitar consultas extras
            // ['compras', 'ventas', 'infoEmpleado'] - Carga compras, ventas e info del empleado
            // ->get() - Ejecuta la consulta y obtiene todos los registros
            
            return response()->json($empleados);
            // Devuelve los empleados en formato JSON
            
        } catch (\Exception $e) {
            // Si hay error, entra aquí
            return response()->json(['error' => $e->getMessage()], 500);
            // Devuelve mensaje de error con código 500 (error del servidor)
        }
    }

    public function store(Request $request)
    {
        // Método para GUARDAR un empleado nuevo
        // Se accede con POST /api/empleados
        
        try {
            $request->validate([
                'nombre_empleado' => 'required|string|max:100',
                // nombre_empleado es obligatorio, debe ser texto, máximo 100 caracteres
                'rol' => 'required|string|max:100',
                // rol es obligatorio, debe ser texto, máximo 100 caracteres
                'fecha_ingreso' => 'required|date',
                // fecha_ingreso es obligatoria, debe ser fecha válida
                'info' => 'nullable|array'
                // info es opcional (nullable), debe ser un array
            ]);

            $empleado = Empleado::create($request->only(['nombre_empleado', 'rol', 'fecha_ingreso']));
            // create() - Inserta nuevo registro en la tabla empleados
            // only() - Solo usa estos campos del request (seguridad)
            
            if (!empty($request->info)) {
                // Si viene información adicional (!empty = NO está vacío)
                InfoEmpleado::create([
                    'id_empleado' => $empleado->id_empleado,
                    'salario' => $request->info['salario'] ?? 0,
                    // ?? 0 - Si no existe, usa 0 como valor por defecto
                    'telefono' => $request->info['telefono'] ?? null,
                    'estado' => $request->info['estado'] ?? 'activo',
                    'fecha_contratacion' => $request->info['fecha_contratacion'] ?? $request->fecha_ingreso
                ]);
            }

            return response()->json($empleado->load('infoEmpleado'), 201);
            // load() - Carga la relación infoEmpleado
            // 201 - Código HTTP para "creado exitosamente"
            
        } catch (\Illuminate\Validation\ValidationException $e) {
            // Si la validación falla
            return response()->json(['error' => $e->errors()], 422);
            // 422 - Código HTTP para "error de validación"
        } catch (\Exception $e) {
            // Si hay otro error
            return response()->json(['error' => $e->getMessage()], 500);
        }
    }

    public function show($id)
    {
        // Método para MOSTRAR un empleado específico
        // Se accede con GET /api/empleados/{id}
        
        $empleado = Empleado::with(['compras', 'ventas', 'infoEmpleado'])->findOrFail($id);
        // findOrFail() - Busca por ID, si no encuentra, lanza error 404
        
        return response()->json($empleado);
        // Devuelve el empleado encontrado
    }

    public function update(Request $request, $id)
    {
        // Método para ACTUALIZAR un empleado
        // Se accede con PUT /api/empleados/{id}
        
        $empleado = Empleado::findOrFail($id);
        // Busca el empleado a actualizar
        
        $request->validate([
            'nombre_empleado' => 'required|string|max:100',
            'rol' => 'required|string|max:100',
            'fecha_ingreso' => 'required|date',
            'info' => 'nullable|array'
        ]);

        $empleado->update($request->only(['nombre_empleado', 'rol', 'fecha_ingreso']));
        // update() - Actualiza el registro con los nuevos datos

        if (!empty($request->info)) {
            $info = InfoEmpleado::firstOrNew(['id_empleado' => $empleado->id_empleado]);
            // firstOrNew() - Busca registro, si no existe, crea uno nuevo
            
            $info->salario = $request->info['salario'] ?? $info->salario ?? 0;
            $info->telefono = $request->info['telefono'] ?? $info->telefono;
            $info->estado = $request->info['estado'] ?? $info->estado ?? 'activo';
            $info->fecha_contratacion = $request->info['fecha_contratacion'] ?? $info->fecha_contratacion ?? $request->fecha_ingreso;
            $info->save();
            // save() - Guarda los cambios en la base de datos
        }

        return response()->json($empleado->load('infoEmpleado'));
    }

    public function destroy($id)
    {
        // Método para ELIMINAR un empleado
        // Se accede con DELETE /api/empleados/{id}
        
        $empleado = Empleado::findOrFail($id);
        $empleado->delete();
        // delete() - Elimina el registro de la base de datos
        
        return response()->json(null, 204);
        // 204 - Código HTTP para "sin contenido" (eliminado exitosamente)
    }
}

<?php

namespace App\Models;
// Espacio de trabajo para modelos

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;
// Importamos clases necesarias de Eloquent (ORM de Laravel)

class Empleado extends Model
{
    use HasFactory;
    // Trait que permite usar factories para datos de prueba

    protected $primaryKey = 'id_empleado';
    // Define que la clave primaria se llama 'id_empleado' (no 'id')
    
    protected $table = 'empleados';
    // Define que la tabla se llama 'empleados' (no 'employees')
    
    protected $fillable = ['nombre_empleado', 'rol', 'fecha_ingreso'];
    // Campos que se pueden llenar masivamente (seguridad contra mass assignment)

    public function compras()
    {
        // Relación: Un empleado tiene MUCHAS compras
        //hasMany() - Método de Eloquent para relación uno a muchos
        
        return $this->hasMany(Compra::class, 'id_empleado');
        // Compra::class - Modelo relacionado
        // 'id_empleado' - Clave foránea en la tabla compras
    }

    public function ventas()
    {
        // Relación: Un empleado tiene MUCHAS ventas
        
        return $this->hasMany(Venta::class, 'id_empleado');
        // Venta::class - Modelo relacionado
        // 'id_empleado' - Clave foránea en la tabla ventas
    }

    public function infoEmpleado()
    {
        // Relación: Un empleado tiene UNA info adicional
        //hasOne() - Método de Eloquent para relación uno a uno
        
        return $this->hasOne(InfoEmpleado::class, 'id_empleado');
        // InfoEmpleado::class - Modelo relacionado
        // 'id_empleado' - Clave foránea en la tabla info_empleados
    }
}

========================================
CONCEPTOS CLAVE EXPLICADOS
========================================

NAMESPACE:
- Organiza las clases para evitar conflictos de nombres
- App\Http\Controllers significa: App → Http → Controllers

USE:
- Importa clases para no escribir la ruta completa
- use App\Models\Empleado; permite escribir Empleado en lugar de App\Models\Empleado

REQUEST:
- Objeto que contiene todos los datos enviados por HTTP
- $request->input('campo') obtiene un campo específico
- $request->all() obtiene todos los campos

VALIDACIÓN:
- $request->validate() verifica que los datos cumplan reglas
- required = obligatorio, string = texto, max:100 = máximo 100 caracteres
- Si falla, lanza ValidationException

RESPONSE:
- response()->json() convierte datos a formato JSON
- response()->json($datos, 200) envía datos con código HTTP 200
- Códigos comunes: 200 (OK), 201 (Creado), 404 (No encontrado), 422 (Error validación), 500 (Error servidor)

MODELOS Y ELOQUENT:
- Model representa una tabla de la base de datos
- Eloquent es el ORM (mapeo objeto-relacional) de Laravel
- Permite trabajar con bases de datos usando objetos en lugar de SQL

RELACIONES:
- hasMany() = uno a muchos (empleado → muchas compras)
- hasOne() = uno a uno (empleado → una info adicional)
- with() carga relaciones para evitar consultas N+1

TRY-CATCH:
- try = intenta ejecutar código que podría fallar
- catch = atrapa errores y los maneja
- Evita que la aplicación se rompa

FILLABLE:
- Protege contra asignación masiva
- Solo los campos en $fillable pueden ser asignados con create() o update()

PRIMARY KEY:
- Clave única que identifica un registro
- Por defecto Laravel usa 'id', pero podemos cambiarla

TABLE:
- Nombre de la tabla en la base de datos
- Por defecto Laravel usa plural del modelo (empleados para Empleado)

========================================
FLUJO COMPLETO DE UNA PETICIÓN
========================================

1. Cliente hace POST a /api/empleados
2. Laravel enruta a EmpleadoController@store
3. Request llega con datos JSON
4. validate() verifica los datos
5. create() inserta en tabla empleados
6. Si hay info, create() inserta en info_empleados
7. load() carga la relación
8. response()->json() devuelve respuesta JSON
9. Cliente recibe respuesta con código 201

<script>
// Inicia bloque de código JavaScript

let clienteActual = null;
let carrito = [];
let productos = [];
// Variables globales para almacenar datos

async function cargarProductos() {
    // async = función asíncrona (puede esperar)
    
    try {
        const response = await fetch('/api/productos');
        // fetch() = hace petición HTTP a la API
        // await = espera a que la petición termine
        // '/api/productos' = endpoint que devuelve productos
        
        productos = await response.json();
        // response.json() = convierte respuesta a JSON
        // await = espera a que se complete la conversión
        
        mostrarProductos();
        // Llama función para mostrar productos en pantalla
        
    } catch (error) {
        console.error('Error:', error);
        // Si hay error, lo muestra en consola
    }
}

function mostrarProductos() {
    const container = document.getElementById('lista-productos');
    // getElementById() = obtiene elemento por su id
    
    container.innerHTML = productos.map(p => `
        <div class="border-2 border-amber-200 rounded-lg p-4">
            <h3>${p.nombre_pan}</h3>
            <p class="text-amber-600 font-bold">$${p.precio}</p>
            <button onclick="agregarAlCarrito(${p.id_pan})">
                Agregar
            </button>
        </div>
    `).join('');
    // productos.map() = transforma cada producto a HTML
    // `` = template literals (permite variables con ${})
    // ${p.nombre_pan} = inserta valor de la propiedad nombre_pan
    // .join('') = une todos los elementos del array en un string
    // innerHTML = inserta HTML en el contenedor
}

async function buscarOCrearCliente() {
    const nombre = document.getElementById('nombre-cliente').value.trim();
    // .value = obtiene valor del input
    // .trim() = quita espacios al inicio y final
    
    if (!nombre) {
        // !nombre = si nombre está vacío o es null/undefined
        alert('Por favor escribe tu nombre');
        return;
        // return = sale de la función
    }
    
    try {
        const response = await fetch('/api/clientes');
        const clientes = await response.json();
        
        clienteActual = clientes.find(c => c.nombre && c.nombre.toLowerCase() === nombre.toLowerCase());
        // .find() = busca primer elemento que cumpla condición
        // c.nombre.toLowerCase() === nombre.toLowerCase() = compara sin importar mayúsculas
        
        if (!clienteActual) {
            // Si no encontró cliente, crea uno nuevo
            
            const crearResponse = await fetch('/api/clientes', {
                method: 'POST',
                // method: 'POST' = indica que vamos a crear
                headers: {
                    'Content-Type': 'application/json',
                    'X-CSRF-TOKEN': '{{ csrf_token() }}'
                    // CSRF token = protección contra ataques
                },
                body: JSON.stringify({ nombre_cliente: nombre })
                // body = datos que enviamos
                // JSON.stringify() = convierte objeto a texto JSON
            });
        }
    } catch (error) {
        console.error('Error:', error);
    }
}

function agregarAlCarrito(idPan) {
    const producto = productos.find(p => p.id_pan === idPan);
    // Busca el producto por su id
    
    const item = carrito.find(c => c.id_pan === idPan);
    // Busca si ya está en el carrito
    
    if (item) {
        // Si ya está, aumenta cantidad
        item.cantidad++;
    } else {
        // Si no está, lo agrega
        carrito.push({
            id_pan: idPan,
            nombre: producto.nombre_pan,
            precio: producto.precio,
            cantidad: 1
        });
        // .push() = agrega elemento al array carrito
    }
    
    actualizarCarrito();
    // Llama función para actualizar visualización
}

HTML:

JSON (JavaScript Object Notation):

HTTP:

- Códigos: 200 (OK), 201 (Creado), 404 (No encontrado), 500 (Error)
